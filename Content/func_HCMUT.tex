\subsubsection{Đối với HCMUT\_DATACORE} 
\begin{itemize} 
    \item Cung cấp dữ liệu người dùng: Họ tên, MSSV, MCB, khoa, email, tình trạng học tập/giảng dạy,.... 
    \item Đồng bộ dữ liệu định kỳ với hệ thống.  
    \item Hỗ trợ truy vấn dữ liệu người dùng theo khóa định danh (MSSV/MCB). 
\end{itemize} 

\subsubsection{Đối với HCMUT\_SSO} 
\begin{itemize} 
    \item Xác thực đăng nhập. 
    \item Quản lý phiên đăng nhập.
    \item Cấp quyền truy cập theo vai trò: Sinh viên, tutor, điều phối viên. 
    \item Thông báo lỗi khi xác thực thất bại và từ chối truy cập. 
    \item Gửi cảnh báo qua email học vụ trong trường hợp đăng nhập thất bại nhiều lần hoặc cần reset mật khẩu. 
\end{itemize} 

\subsubsection{Đối với HCMUT\_Library} 
\begin{itemize} 
    \item Liên kết với các khóa học để cung cấp tài liệu số, giáo trình, bài giảng. 
    \item Hỗ trợ tutor truy cập, chia sẻ và cập nhật tài liệu học tập.
    \item Hỗ trợ sinh viên truy cập tài liệu.
    \item Ghi nhận và thống kê lượt truy cập/tải tài liệu. 
\end{itemize}
